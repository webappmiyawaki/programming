SQL

EMPLOYEEテーブルを作成、レコードを追加、テーブルを削除するには次のSQLを使用してください。

-- EMPLOYEEテーブルを作成するSQL
CREATE TABLE EMPLOYEE (
ID CHAR(6) PRIMARY KEY,
NAME VARCHAR(100) NOT NULL,
AGE INT NOT NULL
);

-- EMPLOYEEテーブルにレコードを追加するSQL
INSERT INTO EMPLOYEE (ID, NAME, AGE) VALUES ('EMP001', '湊 雄輔', 23);
INSERT INTO EMPLOYEE (ID, NAME, AGE) VALUES ('EMP002', '綾部 みゆき', 22);

-- EMPLOYEEテーブルを削除するSQL(やり直し用）
DROP TABLE EMPLOYEE;
